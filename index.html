        function loadData() {

            const textbook = document.getElementById('textbook').value;

            const unitSelect = document.getElementById('unit');

            const data = textbookDB[textbook];

            

            unitSelect.innerHTML = '<option value="">请选择单元</option>';

            if (data) {

                data.units.forEach(unit => {

                    const option = document.createElement('option');

                    option.value = JSON.stringify(unit);

                    option.textContent = `Unit ${unit.id} - ${unit.title}`;

                    unitSelect.appendChild(option);

                });

            }

        }


        function generatePoster() {

            const unitSelect = document.getElementById('unit');

            if (!unitSelect.value) return;

            

            const unit = JSON.parse(unitSelect.value);

            const textbook = document.getElementById('textbook');

            const textbookName = textbook.options[textbook.selectedIndex].text;

            

            // 生成短文

            const templateList = templates[unit.template] || templates.school;

            let passage = templateList[Math.floor(Math.random() * templateList.length)];

            

            // 替换单词

            unit.words.forEach((word, index) => {

                passage = passage.replace(`{w${index+1}}`, `<span class="sentence-highlight">${word.en}</span>`);

            });

            

            // 更新界面

            document.getElementById('poster-meta').textContent = `${textbookName} · Unit ${unit.id}`;

            document.getElementById('poster-title').textContent = unit.title;

            

            // 单词列表

            const wordList = document.getElementById('word-list');

            wordList.innerHTML = '';

            unit.words.forEach(word => {

                const tag = document.createElement('span');

                tag.className = 'word-tag';

                tag.innerHTML = `<b>${word.en}</b><br><small style="color:#666">${word.cn}</small><br><small style="color:#999;font-size:10px">${word.py}</small>`;

                wordList.appendChild(tag);

            });

            

            // 短文

            document.getElementById('passage-content').innerHTML = `<p class="text-xl leading-loose">${passage}</p>`;

            

            // 句型

            const sentenceList = document.getElementById('sentence-list');

            sentenceList.innerHTML = '';

            unit.sentences.forEach(sent => {

                const li = document.createElement('li');

                li.innerHTML = `<span class="text-orange-500 mr-2">▸</span><b>${sent}</b>`;

                sentenceList.appendChild(li);

            });

        }


        function downloadPoster() {

            const poster = document.getElementById('poster');

            html2canvas(poster, {scale: 2}).then(canvas => {

                const link = document.createElement('a');

                link.download = `英语早读海报_${new Date().toLocaleDateString()}.png`;

                link.href = canvas.toDataURL();

                link.click();

            });

        }
